```shell
vim zoo.cfg
#----------------------------------修改zookeeper配置文件--------------------------------------------------
dataDir=/root/data/zookeeper

server.0=192.168.1.128:2888:3888
server.1=192.168.1.129:2888:3888
server.2=192.168.1.130:2888:3888
#--------------------------------------------------------------------------------------------------------


# 在 dataDir 目录下创建 myid 文件， 文件内容为 配置文件中服务器序号； eg：节点机器的ip为 192.168.1.129，
# 那么根据配置文件设置的， myid 文件内容为 1


#----------------------------------环境变量设置-----------------------------------------------------------
export ZK_HOME=/root/services/zookeeper
export PATH=$PATH:$ZK_HOME/bin
#--------------------------------------------------------------------------------------------------------


# 验证集群是否部署成功
zkServer.sh status
```
